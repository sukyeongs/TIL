def dfs(graph, v, visited):
		# ν„μ¬ λ…Έλ“ λ°©λ¬Έμ²λ¦¬
		visited[v] = True
		print(v, end=' ')
		# ν„μ¬ λ…Έλ“μ™€ μ—°κ²°λ λ…Έλ“λ¥Ό μ¬κ·€μ μΌλ΅ λ°©λ¬Έ
		for i in graph[v]:
